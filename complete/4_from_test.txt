<?php
$a=[1,3,5,6,9,10,33];
$b=[2,4,7,8,45,50,66,71,77];

$i=0;
$j=0;
$c=[];
while($i<count($a) and $j<count($b)) {
		if($a[$i]<$b[$j]){
			$c[]=$a[$i];
			$i=$i+1;
		}
		else{
			$c[]=$b[$j];
			$j=$j+1;
		}
}

if(count($a)<count($b)){
	for($j=$j; $j<count($b); $j++){
	$c[]=$b[$j];
	}
}
else{
	for($i=$i; $i<count($a); $i++){
	$c[]=$a[$i];
	}
}

echo '<br>';
print_r ($c);

?>
